\subsection{Verteilungsfunktionen}

\subsubsection{Bernoulli-Verteilung}

\noindent Wird verwendet, wenn es nur 2 Möglichkeiten als Ausgang gibt (Sieg / Niederlage, Gewin / Verlust usw.), z.B. Münzwurf, Los ziehen. \newline \newline
f(x) = $\left\{\begin{array}{cl}
1 - \pi & $für x = 0$\\
\pi & $für x = 1$\\
0 & sonst
\end{array}\right\}$
\newline \newline
E(X) = $\sum\limits_{i=1}^2 x_ip_i = 0*P(X= 0) + 1* P(X=1) = \pi$
\newline \newline
Var(X) = $\sum\limits_{i=1}^2 (x_i - \mu_x)^2p_i = (0 - \pi)^2(1-\pi) + (1 - \pi)^1 \pi = \pi(1 - \pi)$

\subsubsection{Binomialverteilung}

\noindent X sind die Anzahl der Treffer bei n Versuchen, z.B. Multiple Choice \newline \newline
$X_i$ = $\left\{\begin{array}{cl}
1 & $falls im iten Versuch ein Treffer erzielt wird$ \\
0 & $sonst$ \\
\end{array}\right\}$ \newline
und es gilt X = $\sum\limits_{i=1}^n X_i$
 
f(x$|n, \pi$) = $\left\{\begin{array}{cl}
\left(\begin{array}{c}
n\\
x
\end{array}\right)\pi^x(1 - \pi)^{n-x} & $für x = 0,1,2,...,n$ \\
0 & sonst
\end{array}\right\}$
\newline \newline
E(X) = $n * \pi$ 
\newline \newline
Var(X) =$n * \pi * (1- \pi)$

\subsubsection{Diskrete Gleichverteilung}

f(x$|$n) = $\left\{\begin{array}{cl}
\frac{1}{n} & $für x = 1,2,...,n$ \\
0 & $sonst$ \\
\end{array}\right\}$ 
\newline \newline
E(X) = $\frac{n + 1}{2}$
\newline \newline
Var(X) = $\frac{n^2 - 1}{12}$

\subsubsection{Geometrische Verteilung}

\noindent Ein Bernoulli-Experiment (0/1-Verteilung) wird immer wieder wiederholt. Die Zufallsvariable X ist die Versuchsnummer des ersten Treffers.

f(n$|\pi$) = $\left\{\begin{array}{cl}
(1-\pi)^{n-1}*\pi & $für n = 1,2,3,...$ \\
0 & $sonst$ \\
\end{array}\right\}$ 
\newline \newline
E(X) = $\frac{1}{\pi}$
\newline \newline
Var(X) = $\frac{1 - \pi}{\pi^2}$

\subsubsection{Hypergeometrische Verteilung}

\noindent M aus N Elementen haben eine bestimmte Eigenschaft (Losgewinn). 
\newline Wenn n Elemente mit Zurücklegen gezogen werden:
$\pi = \frac{M}{N}$
\newline 
\newline Wenn n Elemente ohne Zurücklegen gezogen werden, ist die Anzahl X nicht mehr exakt binomialverteilt.

f(n$|$n,N,M) = $\left\{\begin{array}{cl}
\frac{ 
\left(\begin{array}{c}
M \\
x \\
\end{array}\right) * 
\left(\begin{array}{c}
N-M \\
n-x \\
\end{array}\right) }
{
\left(\begin{array}{c}
N \\
n \\
\end{array}\right) } 
& $für x = max(0,n - (N-M)),..., min(n,M)$ \\
0 & $sonst$ \\
\end{array}\right\}$ 
\newline \newline
E(X) = $n*\frac{M}{N}$
\newline \newline
Var(X) = $n*\frac{M}{N}*\left(1 - \frac{M}{N}\right)*\frac{N-n}{N-1}$


\subsubsection{Poisson-Verteilung}

\noindent Muss in der Klausur erwähnt werden, kann nicht selbst entschieden werden.
\newline z.B. Anrufe in best. Zeitabschnitt in einer Hotline oder Anzahl der Personen, die in einem best. Zeitabschnitt einen Schalter besuchen

f(x$|\lambda$) = $\left\{\begin{array}{cl}
\frac{\lambda^x}{x!}\epsilon^{-\lambda} & $für x = 0,1,2,3,...$ \\
0 & $sonst$ \\
\end{array}\right\}$ 
\newline \newline
E(X) = $\lambda$
\newline \newline
Var(X) = $\lambda$


$(n+1)! = (n+1)*n! \leq (n+1)*n^n = n^{n+1} + n^n \leq (n+1)^{n+1}$

\subsection{Normalverteilung}
Gegeben ist $\mu = E(X)$, sowie $\sigma^2 = Var(X)$, z.B.: N($E(X)$, $Var(x)$) \newline
Dies kann auf die Standardnormalverteilung $\mu = 0$, sowie $\sigma = 1$ zurückgerechnet werden: \newline
$P(z = \frac{x - \mu}{\sigma}) = z_p$


\subsubsection{Rechenbeispiel - Normalverteilung}
N($4000$, $10^6$)
\newline Wie hoch ist die Wahrscheinlichkeit für weniger als 3000? \newline
\begin{table}[!h]
\begin{tabular}{rl}
$P(Z\leq \frac{3000 - 4000}{10^3})$ & $= z_p$ \\
$P(Z \leq -1)$ & $= 1 - P(Z \geq 1)$ \\
 & $= 1 - 0.84$ \\
 & $= 0.16$
\end{tabular}
\end{table}
